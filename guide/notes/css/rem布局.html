# 1.2 flexible.js + rem

max-width: 750px;不可以省略 因为flexible是根据当前屏幕调整分为10等份，所以需要限定

body {
  min-width: 320px;
  /* 因为flexible是根据当前屏幕调整分为10等份，所以需要限定 */
  max-width: 750px;
  /* flexible划分为10等份 */
  width: 10rem;
  margin: 0 auto;
  line-height: 1.5;
  font-family: Arial, Helvetica;
  background: #f2f2f2;
}
